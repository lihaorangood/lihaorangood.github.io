# typeof instanceof

# 简介

typeof 的用法想必大家都比较熟悉，一般被用于来判断一个变量的类型。我们可以使用 typeof 来判断 number、undefined、symbol、string、function、boolean、object 这七种数据类型。但是遗憾的是，typeof 在判断 object 类型时候，有些许的尴尬。它并不能明确的告诉你，该 object 属于哪一种 object。所以我们最好使用typeof进行判断简单数据类型， 而不是使用 typeof 判断 null 或者 Array 等无法给出具体数据类型的数据

instanceof 和 typeof 非常的类似。instanceof 运算符用来检测 constructor.prototype 是否存在于参数 object 的原型链上。与 typeof 方法不同的是，instanceof 方法要求开发者明确地确认对象为某特定类型。

function C() {}
const c = new C();
// c.constructor == C  c.__proto__ == C.prototype

// instanceof 实现原理
function instance_of(L, R) {
  //L 表示左表达式，R 表示右表达式
  var O = R.prototype; // 取 R 的显示原型
  L = L.__proto__; // 取 L 的隐式原型
  while (true) {
    if (L === null) return false;
    if (O === L)
      // 这里重点：当 O 严格等于 L 时，返回 true
      return true;
    L = L.__proto__;
  }
}

# 继承 （标记）

  • 类式继承
  • 构造继承
  • 组合式继承
// 类式继承 无法共享变量 且 无法向父类创建实例 传递参数 变量 共享 这种最重要的确定啊
function SuperClass() {
  this.superVal = true;
}
SuperClass.prototype.isSuperClass = true;
SuperClass.prototype.sayHi = function() {
  console.log(this.superVal);
};
function SubClass() {
  this.subval = false;
}
SubClass.prototype = new SuperClass();
SubClass.prototype.saySubval = function() {
  console.log(this.subval);
};
const instance = new SuperClass();
console.log(instance instanceof SuperClass); // true
console.log(instance instanceof SubClass); // true
console.log(SubClass instanceof SuperClass); // true
// 解析 instance.__proto__  == SubClass.prototype  第一个 为 TRUE
// SubClass.prototype.__proto__ = SuperClass.prototype 第二个为 TRUE 第三个为 false

// 构造函数继承 继承变量 独立 但是 方法无法共享 此为去缺点
function SuperClass(id) {
  this.books = ["js", "css"];
  this.id = id;
}
SuperClass.prototype.showBooks = function() {
  console.log(this.books);
};
function SubClass(id) {
  //继承父类
  SuperClass.call(this, id);
}
//创建第一个子类实例
var instance1 = new SubClass(10);
//创建第二个子类实例
var instance2 = new SubClass(11);

instance1.books.push("html");
console.log(instance1);
console.log(instance2);
instance1.showBooks(); //TypeError

// 组合式继承 将上述两种方法结合 解决问题 但是 SuperClass执行了两次
function SuperClass(name) {
  this.name = name;
  this.books = ["Js", "CSS"];
}
SuperClass.prototype.getBooks = function() {
  console.log(this.books);
};
function SubClass(name, time) {
  SuperClass.call(this, name);
  this.time = time;
}
SubClass.prototype = new SuperClass();

SubClass.prototype.getTime = function() {
  console.log(this.time);
};
