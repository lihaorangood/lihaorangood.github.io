# init 函数

在 init 函数中我们需要执行 合并参数值全局的一些参数合并至当前实例中 比如全局组件与私有组件的合并 执行一些钩子函数比如 created, 最主要的是执行一些初始化, 比如 initState 在 initState 初始化中我们进行 props data methods watch 的校验、创建和代理 所以我们才能通过 this 也就是组件实例进行访问定义的属性和方法 在 methods 的方法中已经绑定了 this 为当前组件实例 不可更改

Vue.prototype._init = function(options) {
  const vm = this
  vm.$options = mergeOptions(vm.constructor.options, options) // 需要将用户自定义的options和全局的options做合并
  // props data watch computed watch methods不同的处理
  callHook(vm, 'beforeCreate')
  initState(vm)
  callHook(vm, 'created')
  if (vm.$options.el) {
    vm.$mount(vm.$options.el)
  }
}

# mount 函数

编译 template 获取组件的 render || template|| outHTML 最终都会转换成 render 函数 通过 render 函数 生成 ast 然后生成 VDomTree, 通过 patch 方法挂载到 我们设定的指定元素上

Vue.prototype.$mount = function(el) {
  if (!el) {
    debugger
  }
  el = document.querySelector(el)
  const vm = this
  vm.$el = el
  const options = vm.$options
  if (!options.render) {
    let template = options.template
    if (!template && el) {
      template = el.outerHTML
    }
    // 将模板生成render函数 渲染都是使用render函数
    options.render = compileToFunctions(template)
    //    挂载组件
    callHook(vm, 'beforeMount')
    mountComponent(vm, el)
  }
}

# 响应式原理以及 watch 的应用 (渲染, watcher 用户创建, computed)

响应式原理其实是将数据和视图关联起来 通过代理知道什么时候数据改变然后知道什么时候同时视图 data 通过代理 和 dep 关联起来 dep 和 watcher 关联起来 watcher 被 dep 手机 然后触发 watcher 中的回调更新视图 watcher 其实有三类 渲染 watcher VueJS 帮我们自动建立一个组件更新 watcher 然后我们可以手动观测数据 computed 也是一个 watcher 数据 watcher 可以通过 immediate 属性立即执行和变化自动执行 computed 只能通过有取值时被动执行 但最终调用的还是$watch 方法

// watcher
export class Watcher {
  constructor(vm, exprOrFn, callback, options) {
    this.vm = vm
    this.exprOrFn = exprOrFn
    this.callback = callback
    this.options = options
    this.id = id++ // watcher 的唯一标识
    this.deps = [] // 记录dep 实现双向记录
    this.isWatcher = !options.user // 标记是否渲染watcher 还是用户自主创建的 监听watcher
    this.depsId = new Set()
    if (options) {
      this.user = !!options.user
      this.sync = !!options.sync
      this.deep = !!options.deep
      this.immediate = !!options.immediate
      this.lazy = !!options.lazy
      this.dirty = this.lazy // 初始值为false 控制computed取值是否重新执行
    } else {
      this.user = this.sync = this.deep = this.immediate = false
    }
    if (typeof exprOrFn === 'function') {
      this.getter = exprOrFn
    } else {
      this.getter = function() {
        const path = exprOrFn.split('.')
        let obj = vm
        obj = path.reduce((newVal, currentVal) => newVal[currentVal], obj)
        return obj
      }
    }
    this.value = this.lazy ? void 0 : this.get()
    if (this.immediate) {
      this.callback(this.value)
    }
    if (this.deep) {
    }
  }

  addDep(dep) {
    const id = dep.id
    if (!this.depsId.has(id)) {
      this.deps.push(dep)
      this.depsId.add(id)
      dep.addWatcher(this)
    }
  }

  get() {
    // 借助JS单线程的机制
    pushTarget(this)
    const result = this.getter.call(this.vm) // 调用exprOrFn 渲染界面 取值 记性patch _render => with(this){_v(msg)***}
    // 深度监听处理 递归遍历所有属性 将watch和d属性dep建立练习
    if (this.deep) {
      traverse(result)
    }
    popTarget()
    return result
  }

  run() {
    //  只有渲染watcher才会调用更新hooks函数
    if (this.isWatcher) {
      callHook(this.vm, 'beforeUpdate')
    }
    const newVal = this.get()
    const oldVal = this.value
    this.callback.call(this.vm, newVal, oldVal)
    this.value = newVal
  }

  evaluate() {
    this.value = this.get()
    this.dirty = false // 置为FALSE 防止数据不改变 不重新计算和执行
  }

  depend() {
    let i = this.deps.length
    while (i--) {
      this.deps[i].depend()
    }
  }

  update() {
    if (this.lazy) {
      this.dirty = true
    } else {
      // 这里不要每一次都更新 我们需要做暂存异步更新 批量更新
      queueWatcher(this)
      // this.get()
    }
  }
}

// dep
let id = 0
const watcherStack = []
// 用栈存储watcher  防止有计算属性 依赖 但是依赖没有在视图上渲染  计算属性渲染了  所以我们就需要先存储一下渲染watcher
// 然后让计算属性的依赖在存储计算watcher的同时还存储渲染watcher用来更改依赖只会更新视图
/**
 * 为什么先添加渲染watcher ？？？
 * 首先如果依赖没有 在视图上做渲染 计算属性做渲染了 按照代码创建顺序是 计算属性先创建 然后是渲染watcher.get执行添加到占中
 * 但是只有在渲染watcher创建之后进行取值的时候 才会取依赖蜀兴的值 然后栈才添加计算属性的watcher
 */

/**
 * Dep 和watcher 是多对多的关系 因为每一个组件都有一个watcher实例 每一个组件的每一个属性都有一个dep实例
 * 看起来是dep和watcher对一对一的关系 但是用户可以通过 $watch方法生成一个新的watcher
 * 一个watcher可以对应多个Watcher
 * 连接了数据的get set
 */
export class Dep {
  constructor() {
    this.watcherSubs = []
    this.id = id++
  }

  depend() {
    // 我们希望watcher 也能记住每一个dep

    Dep.target.addDep(this)
  }

  addWatcher(watcher) {
    this.watcherSubs.push(watcher)
  }

  notify() {
    this.watcherSubs.forEach((wather) => {
      wather.update()
    })
  }
}

Dep.target = undefined

export function pushTarget(watcher) {
  Dep.target = watcher
  watcherStack.push(watcher)
}

export function popTarget() {
  watcherStack.pop()
  Dep.target = watcherStack[watcherStack.length - 1]
}

// observer
import { arrayMethods } from './array'
import { Dep } from './dep'

class Observer {
  constructor(data) {
    this.dep = new Dep()
    Object.defineProperty(data, '__ob__', {
      enumerable: false, // 不能被枚举 无法被循环到该属性
      configurable: false,
      value: this
    })
    if (Array.isArray(data)) {
      // 不代理数据数组中的数据 数组中的数据可能会有很多个 如果有很多个数据的话不停的添加set和get 会造成卡顿
      data.__proto__ = arrayMethods
    } else {
      this.walk(data)
    }
  }

  observeArray(arr) {
    arr.forEach((item) => {
      observer(item)
    })
  }

  walk(data) {
    const keys = Object.keys(data)
    keys.forEach((key) => {
      defineReactive(data, key, data[key]) // Vue.util.defineReactive 有这个方法
    })
  }
}

/**
 * 每次调用形成闭包 value的值得以保存
 * @param obj
 * @param key
 * @param value
 * @returns {*}
 */
function defineReactive(obj, key, value) {
  const childObserverInstance = observer(value)
  const dep = new Dep()
  Object.defineProperty(obj, key, {
    get() {
      if (Dep.target) {
        if (childObserverInstance) {
          childObserverInstance.dep.depend()
        }
        dep.depend()
      }
      return value
    },
    set(newValue) {
      if (newValue === value) return
      observer(newValue)
      value = newValue
      dep.notify()
    }
  })
}

export function observer(data) {
  //     进行判断data应该是个对象 并且data！== null
  if (typeof data !== 'object' || data === null) {
    return
  }
  if (data.__ob__) {
    return data // 防止重复观测
  }
  return new Observer(data)
}

// hackarray方法 因为数据有索引 数据量可能也非常多 为索引建立响应式是非常耗费性能的一件事 并且使用数据方法时非常有可能造成
// 数组塌陷 造成重新排列
const oldArrayProtoMethods = Array.prototype
const methods = ['push', 'pop', 'shift', 'unshift', 'sort', 'reverse', 'splice']

export const arrayMethods = Object.create(oldArrayProtoMethods)
methods.forEach((method) => {
  arrayMethods[method] = function(...args) {
    const ob = this.__ob__
    const result = oldArrayProtoMethods[method].apply(this, args)
    // 通知视图进行更新
    let inserted
    switch (method) {
      case 'push':
      case 'unshift':
        inserted = args
        break
      case 'splice':
        inserted = args.slice(2) //arr.splice(0,1, {a: 1})
        break
      default:
        break
    }
    if (inserted) {
      ob.observeArray(inserted)
    }
    ob.dep.notify()
    return result
  }
})
